181 void interrupt_call_receiver() {
182  /* Get a request message */
183  if ((r = driver_receive(ANY, &msg, &ipc_status)) != 0) {
184  printf("driver_receive failed with: %d", r);
185  return;
186  }
187  if (is_ipc_notify(ipc_status)) { /* received notification */
188  switch (_ENDPOINT_P(msg.m_source)) {
189  case HARDWARE: /* hardware interrupt notification */
190  if (msg.m_notify.interrupts & get_irq_set(TIMER)) {
191  timer_handler();
192  }
193  if (msg.m_notify.interrupts & get_irq_set(MOUSE)) { /* subscribed interrupt */
194  mouse_handler();
195  }
196  if (msg.m_notify.interrupts & get_irq_set(KBD)) {
197  kbd_handler();
198  }
199  if (msg.m_notify.interrupts & get_irq_set(RTC)) {
200  rtc_handler();
201  }
202  if (msg.m_notify.interrupts & get_irq_set(SER)) {
203  ser_handler();
204  }
205  break;
206  default:
207  break; /* no other notifications expected: do nothing */
208  }
209  }
210  else { /* received a standard message, not a notification */
211  /* no standard messages expected: do nothing */
212  }
213  return;
214 }
