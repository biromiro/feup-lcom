35  uint8_t st = 0, timeout_cnt = 0;
36 
37  /* loops while 8042 input buffer is not empty */
38  do {
39 
40  /* checks if it is possible to read the status of KBC */
41  if (kbc_get_status(&st) != 0) {
42  printf("Error getting KBC status");
43  return -1;
44  }
45 
46  /* checks if the input buffer is empty */
47  if ((KBC_IBF & st) == 0) {
